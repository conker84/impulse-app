# engine speed histogram
hist1_name = "eng_spd_hist_p1"
hist1_bins = [float(i) for i in range(0, 6000, 500)]
hist1_desc = "Distribution of the engine speed over time"
hist1 = Histogram(
    name=hist1_name,
    base_expr=signals["Eng_Spd_masked"],
    bins=hist1_bins,
    desc=hist1_desc,
    agg_type="duration",
    bins_unit="rpm"
)
my_first_page.add_aggregation(hist1)
